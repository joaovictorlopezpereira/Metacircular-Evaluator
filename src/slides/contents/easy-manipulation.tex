
\begin{frame}[fragile]
  \frametitle{Manipulação de Estruturas de Dados}
  É fácil manipular estruturas de dados! (pois todas as estruturas são listas!)
  \vspace{0.5cm}
  \pause
  \begin{code}
  (define map
    (lambda (f lst)
      (if (null? lst)
          '()
          (cons (f (car lst))
                (map f (cdr lst))))))
  \end{code}
\end{frame}
